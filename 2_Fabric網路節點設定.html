Fabric網路節點設定

1. 複製範例「test-network」系統設定文件進行後續設定。

# 複製範例「test-network」系統設定文件
$ cd $HOME
$ mkdir ./fabric-dev
$ cp -a ~/fabric-samples/bin ~/fabric-dev
$ cp -a ~/fabric-samples/test-network/organizations/cryptogen ~/fabric-dev
$ cp -a ~/fabric-samples/test-network/configtx ~/fabric-dev
$ cp -a ~/fabric-samples/test-network/docker/ ~/fabric-dev

2. 新增Chaincode目錄，複製Chaincode範本。後續將以範例asset-transfer-basic進行安裝。

# 新增Chaincode目錄(Chaincode部署所在目錄)
$ cd ~/fabric-dev
$ mkdir ./chaincode

# 複製範例asset-transfer-basic(測試環境才需執行)
$ cp -a ~/fabric-samples/asset-transfer-basic ~/fabric-dev/chaincode

3. 編輯cryptogen目錄下組織與節點的憑證設定文件。

3.1. 編輯crypto-config-orderer.yaml，設置排序節點(Orderer)憑證創建內容。

# OrdererOrgs段落，設定排序(Orderer)節點名稱、域名與數量。
OrdererOrgs:
  - Name: Orderer
    Domain: example.com
    EnableNodeOUs: true
    Specs:
      - Hostname: orderer
      SANS:
        - localhost

3.2. 編輯crypto-config-org1.yaml，設置Org1節點憑證創建內容。

# PeerOrgs段落，設定組織名稱、域名與帳號數量。
# Template.Count：指節點(Peer)數量。
# Users.Count：指除Admin帳號以外帳號數量。
# 此範例後續將生成Admin與User1兩個帳號。
PeerOrgs:
  - Name: Org1
    Domain: org1.example.com
    EnableNodeOUs: true
    Template:
      Count: 1
      SANS:
        - localhost
    Users:
      Count: 1

3.3. 編輯crypto-config-org2.yaml，設置Org2節點憑證創建內容。

# PeerOrgs段落，設定組織名稱、域名與帳號數量。
# Template.Count：指節點(Peer)數量。
# Users.Count：指除Admin帳號以外帳號數量。
# 此範例後續將生成Admin與User1兩個帳號。
PeerOrgs:
  - Name: Org2
    Domain: org2.example.com
    EnableNodeOUs: true
    Template:
      Count: 1
      SANS:
        - localhost
    Users:
      Count: 1

4. 編輯configtx/configtx.yaml，此為生成Channel設置交易來源文件。

# Organizations段落，設定Orderer、Org1與Org2名稱、ID、MSP目錄。新增AnchorPeers設定。
Organizations: 
    - &Org1
        ...略
        AnchorPeers:
            - Host: peer0.org1.example.com
              Port: 7051

    - &Org2
        ...略
        AnchorPeers:
            - Host: peer0.org2.example.com
              Port: 9051

# Capabilities段落，設定Fabric Network版本相容性。此段落不需調整變更。
# Application段落，設定網路應用程式的組織。此段落不需調整變更。
# Orderer段落，設定共識機制、區塊產生條件與各共識機制規格設定。
# 此段落將修改下列設定：
BatchSize:
  MaxMessageCount: 100
  AbsoluteMaxBytes: 32 MB

5. 執行cryptogen 創建節點憑證(正式環境不建議使用)

# 設定fabric tools環境變數
$ export PATH=${PWD}/bin:${PWD}:$PATH

# 測試環境變數是否正確
$ cryptogen --help

# cryptogen讀取crypto-config.yaml創建節點證書
# 執行後，產生crypto-config資料夾，內有各節點證書檔案。
$ rm -Rf organizations/peerOrganizations
$ rm -Rf organizations/ordererOrganizations
$ cryptogen generate --config=./cryptogen/crypto-config-org1.yaml --output="organizations"
$ cryptogen generate --config=./cryptogen/crypto-config-org2.yaml --output="organizations"
$ cryptogen generate --config=./cryptogen/crypto-config-orderer.yaml --output="organizations"

6. 執行configtxgen 生成創世區塊與Channel設置交易檔

# 設定環境變數
# SYS_CHANNEL：系統CHANNEL名稱，不可與CHANNEL_NAME相同。
# CHANNEL_NAME：自訂的CHANNEL名稱，在整個Fabric Network中須是唯一且不可重複。

$ rm -rf system-genesis-block/*.block channel-artifacts/*.tx
$ export FABRIC_CFG_PATH=$PWD/configtx
$ export SYS_CHANNEL=sys-channel
$ export CHANNEL_NAME=dev-channel

# 新增channel-artifacts目錄
$ mkdir ./channel-artifacts

# 生成創世區塊
# system-genesis-block目錄下生成genesis.block檔案
# configtxgen -profile [共識機制配置名稱] -channelID [$SYS_CHANNEL] -outputBlock [生成區塊的路徑與名稱.block]
$ configtxgen -profile TwoOrgsOrdererGenesis -channelID $SYS_CHANNEL -outputBlock ./system-genesis-block/genesis.block

# 生成Channel設置交易檔案
# channel-artifacts目錄下生成channel.tx檔案
# configtxgen -profile [Channel配置名稱] -outputCreateChannelTx  [生成Channel設置交易檔案路徑與名稱.tx] -channelID [$CHANNEL_NAME]
$ export CHANNEL_NAME=dev-channel
$ configtxgen -profile TwoOrgsChannel -outputCreateChannelTx ./channel-artifacts/channel.tx -channelID $CHANNEL_NAME

# 生成各組織錨節點Channel設置交易檔案
# channel-artifacts目錄下生成Org1MSPanchors.tx、Org2MSPanchors.tx檔案
# configtxgen -profile [Channel配置名稱] -outputAnchorPeersUpdate [生成錨節點Channel設置交易檔案路徑與名稱.tx] -channelID [$CHANNEL_NAME] -asOrg [Organizations/Org1名稱]
$ export CHANNEL_NAME=dev-channel

# Org1
$ configtxgen -profile TwoOrgsChannel -outputAnchorPeersUpdate ./channel-artifacts/Org1MSPanchors.tx -channelID $CHANNEL_NAME -asOrg Org1MSP

# Org2
$ configtxgen -profile TwoOrgsChannel -outputAnchorPeersUpdate ./channel-artifacts/Org2MSPanchors.tx -channelID $CHANNEL_NAME -asOrg Org2MSP

啟動Hyperledger Fabric網路

1. 啟動區塊鏈網路

# 啟動區塊鏈網路
# docker-compose -f [docker container compose配置文件] up -d
# 同時啟動多個docker container compose配置文件，續加 -f 參數即可
# 參數 -d 指container背景執行
# 設定docker peer image 版本(2.2.2)
# 設定docker peer-CA image 版本(1.4.9)

$ export FABRIC_CFG_PATH=$PWD
$ export IMAGE_TAG=2.2.2
$ export CA_IMAGE_TAG=1.4.9
$ export SYS_CHANNEL=sys-channel
$ export COMPOSE_PROJECT_NAME=dev
$ docker-compose -f ./docker/docker-compose-test-net.yaml -f ./docker/docker-compose-ca.yaml -f ./docker/docker-compose-couch.yaml up -d 2>&1

2. 停止區塊鏈網路

# 停止區塊鏈網路
# 設定docker peer image 版本(2.2.2)
# 設定docker peer-CA image 版本(1.4.9)

$ export FABRIC_CFG_PATH=$PWD
$ export IMAGE_TAG=2.2.2
$ export CA_IMAGE_TAG=1.4.9
$ export SYS_CHANNEL=sys-channel
$ export COMPOSE_PROJECT_NAME=dev
$ docker-compose -f ./docker/docker-compose-test-net.yaml -f ./docker/docker-compose-ca.yaml -f ./docker/docker-compose-couch.yaml stop

3. 卸載Docker Container 並刪除附屬的Volumes

# 設定docker peer image 版本(2.2.2)
# 設定docker peer-CA image 版本(1.4.9)

$ export FABRIC_CFG_PATH=$PWD
$ export IMAGE_TAG=2.2.2
$ export CA_IMAGE_TAG=1.4.9

$ export COMPOSE_PROJECT_NAME=dev
$ docker-compose -f ./docker/docker-compose-test-net.yaml -f ./docker/docker-compose-ca.yaml -f ./docker/docker-compose-couch.yaml down --volumes --remove-orphans

創建與加入Channel

1. 創建Channel

# 進入docker CLI container執行指令
$ docker exec -it cli bash

# 創建Channel執行指令
# peer channel create -o orderer.example.com:7050 -c [$CHANNEL_NAME] -f [生成Channel設置交易檔案路徑與名稱].tx --tls --cafile [orderer TLS 公鑰檔案路徑與名稱.pem]

# 透過peer0.org1.example.com創建Channel
# 轉換至peer0.org1.example.com
$ export CORE_PEER_LOCALMSPID="Org1MSP"
$ export CORE_PEER_MSPCONFIGPATH=/opt/gopath/src/github.com/hyperledger/fabric/peer/organizations/peerOrganizations/org1.example.com/users/Admin@org1.example.com/msp
$ export CORE_PEER_TLS_ROOTCERT_FILE=/opt/gopath/src/github.com/hyperledger/fabric/peer/organizations/peerOrganizations/org1.example.com/peers/peer0.org1.example.com/tls/ca.crt
$ export CORE_PEER_ADDRESS=peer0.org1.example.com:7051

$ export CHANNEL_NAME=dev-channel
$ peer channel create -o orderer.example.com:7050 -c $CHANNEL_NAME -f ./channel-artifacts/channel.tx --outputBlock ./channel-artifacts/${CHANNEL_NAME}.block --tls --cafile /opt/gopath/src/github.com/hyperledger/fabric/peer/organizations/ordererOrganizations/example.com/orderers/orderer.example.com/msp/tlscacerts/tlsca.example.com-cert.pem

Chaincode安裝

1. 打包Chaincode

# 打包Chaincode指令
# peer lifecycle chaincode package [打包名稱.tar.gz] --path [Chaincode路徑] --lang [Chaincode語言] --label [Chaincode標籤]

# 打包檔案副檔名必須為.tar.gz
# Chaincode語言：golang(default)、node(javascript或typescript)、java
# --label：建議採Chaincode Name + 版本號組合

$ peer lifecycle chaincode package mycc.tar.gz --path /opt/gopath/src/github.com/hyperledger/fabric/peer/chaincode/asset-transfer-basic/chaincode-javascript --lang node --label mycc_1.0
