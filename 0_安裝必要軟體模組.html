Hyperledger Fabric 安裝必要軟體模組

  1. 更新軟體模組套件清單

    # Update package lists
$ sudo apt-add-repository -y ppa:git-core/ppa
$ sudo apt-get update

  2. 安裝CURL

     # Install cURL
 $ sudo apt-get install -y curl

  3. 安裝Git

    # Install Git
$ sudo apt-get install -y git

  4. 安裝unzip

    # Install unzip
$ sudo apt-get -y install unzip

  5. 安裝Docker與Docker Compose

    # Update package lists
$ sudo apt-get update

# Ensure that CA certificates are installed
$ sudo apt-get -y install apt-transport-https ca-certificates gnupg-agent software-properties-common

# Add Docker repository key to APT keychain
$ curl -fsSL https://download.docker.com/linux/ubuntu/gpg | sudo apt-key add -

# Installing Docker
$ sudo add-apt-repository "deb [arch=amd64] https://download.docker.com/linux/ubuntu $(lsb_release -cs) stable"
$ sudo apt-get update
$ sudo apt-get -y install docker-ce docker-ce-cli containerd.io

# Add user account to the docker group
$ sudo usermod -aG docker $(whoami)

# Installing Docker-Compose
$ sudo curl -L "https://github.com/docker/compose/releases/download/1.25.4/docker-compose-$(uname -s)-$(uname -m)" -o /usr/local/bin/docker-compose
$ sudo chmod +x /usr/local/bin/docker-compose

  6. 安裝Node.JS

    # Install NodeJS
$ sudo apt-get -y install build-essential libssl-dev

# Execute nvm installation script
$ curl -o- https://raw.githubusercontent.com/creationix/nvm/master/install.sh | bash

# Set up nvm environment without restarting the shell
$ export NVM_DIR="${HOME}/.nvm"
[ -s "${NVM_DIR}/nvm.sh" ] && . "${NVM_DIR}/nvm.sh"
[ -s "${NVM_DIR}/bash_completion" ] && . "${NVM_DIR}/bash_completion"

# Install node 12.20.2
$ nvm install v12.20.2
$ nvm use v12.20.2

  7. 安裝Go-Lang

    # Installing Go-Lang v1.14.12
$ sudo wget https://dl.google.com/go/go1.14.12.linux-amd64.tar.gz -O "/usr/local/go1.14.12.linux-amd64.tar.gz"
$ sudo tar -C /usr/local -xzf /usr/local/go1.14.12.linux-amd64.tar.gz

# Set up Go environment variables to ~/.profile
$ echo ' 
# Go-lang PATH
export GORPATH="/usr/local/go"
export GOPATH="/home/'$(whoami)'/go"
export PATH="$PATH:$GORPATH/bin"' >> ~/.profile

  8. Ubuntu重新開機

  9. 確認軟體模組安裝結果

    # Print installation details for user
$ docker --version
$ docker-compose --version
$ nvm --version
$ node --version
$ npm --version
$ go version
