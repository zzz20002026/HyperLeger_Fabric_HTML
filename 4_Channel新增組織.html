Channel新增組織

將Org3新增至dev-channel

Note: 作業前需在Fabric Network已啟動運作下進行

1.準備所需檔案

# 複製範例「test-network/addOrg3」系統設定文件
$ cd $HOME
$ mkdir -p ~/fabric-dev/configtx-org3/
$ cp -a ~/fabric-samples/test-network/addOrg3/docker/*.yaml ~/fabric-dev/docker
$ cp -a ~/fabric-samples/test-network/addOrg3/configtx.yaml ~/fabric-dev/configtx-org3/
$ cp -a ~/fabric-samples/test-network/addOrg3/org3-crypto.yaml ~/fabric-dev/cryptogen
$ cp -a ~/fabric-samples/config ~/fabric-dev/config

2.設置Org3 Docker Container環境設定

# docker-compose-org3.yaml：Peer0節點，修改下列設定值。
services:

  peer0.org3.example.com:
  ...以上略...
 volumes:
        - /var/run/docker.sock:/host/var/run/docker.sock
        # 修改憑證路徑
        #- ../../organizations/peerOrganizations/org3.example.com/peers/peer0.org3.example.com/msp:/etc/hyperledger/fabric/msp
        #- ../../organizations/peerOrganizations/org3.example.com/peers/peer0.org3.example.com/tls:/etc/hyperledger/fabric/tls
        - ../organizations/peerOrganizations/org3.example.com/peers/peer0.org3.example.com/msp:/etc/hyperledger/fabric/msp
        - ../organizations/peerOrganizations/org3.example.com/peers/peer0.org3.example.com/tls:/etc/hyperledger/fabric/tls
        - peer0.org3.example.com:/var/hyperledger/production

3.執行cryptogen創建Org3節點憑證

# 將在fabric-dev/organizations/peerOrganizations新增org3
$ cd ~/fabric-dev
$ ./bin/cryptogen generate --config=./cryptogen/org3-crypto.yaml --output="./organizations"

4.創建組織交易檔

# 在fabric-dev/organizations/peerOrganizations/org3.example.com目錄新增org3.json檔案
# 創建過程會讀取fabric-dev/configtx/org3/configtx.yaml

$ cd ~/fabric-dev
$ export FABRIC_CFG_PATH=$PWD/configtx-org3
$ ./bin/configtxgen -printOrg Org3MSP >./organizations/peerOrganizations/org3.example.com/org3.json

5.啟動Org3 Docker Container

# 設定docker peer image 版本(2.2.2)
# 設定docker peer-CA image 版本(1.4.9)

$ export FABRIC_CFG_PATH=$PWD
$ export IMAGE_TAG=2.2.2
$ export CA_IMAGE_TAG=1.4.9
$ export SYS_CHANNEL=sys-channel
$ export COMPOSE_PROJECT_NAME=dev

$ docker-compose -f ./docker/docker-compose-org3.yaml up -d 2>&1

6.獲取dev-channel最新配置區塊並產生二進位的protobuf檔案

# 配置區塊應為Block 2，Block 0為genesis block，Block 1為Org1 anchor peer update，Block 2為Org2 anchor peer update。
# 將在channel-artifacts目錄下新增config_block.pb

$ export PATH=${PWD}/bin:$PATH
$ export FABRIC_CFG_PATH=${PWD}/config/
$ export CORE_PEER_TLS_ENABLED=true
$ export CORE_PEER_LOCALMSPID="Org1MSP"
$ export CORE_PEER_TLS_ROOTCERT_FILE=${PWD}/organizations/peerOrganizations/org1.example.com/peers/peer0.org1.example.com/tls/ca.crt
$ export CORE_PEER_MSPCONFIGPATH=${PWD}/organizations/peerOrganizations/org1.example.com/users/Admin@org1.example.com/msp
$ export CORE_PEER_ADDRESS=localhost:7051

$ peer channel fetch config channel-artifacts/config_block.pb -o localhost:7050 --ordererTLSHostnameOverride orderer.example.com -c dev-channel --tls --cafile ${PWD}/organizations/ordererOrganizations/example.com/orderers/orderer.example.com/msp/tlscacerts/tlsca.example.com-cert.pem

7.轉換配置區塊為JSON格式與擷取配置內容

# configtxlator工具程式用於將區塊資料在protobuf和JSON格式間轉換，以及調整配置。
# jq是操作JSON檔案的工具程式

# 安裝jq
$ sudo apt install jq -y

# 將config_block.pb轉換為config.json
$ configtxlator proto_decode --input ./channel-artifacts/config_block.pb --type common.Block --output ./channel-artifacts/config_block.json
$ jq .data.data[0].payload.data.config ./channel-artifacts/config_block.json > ./channel-artifacts/config.json

8.加入Org3的憑證資料

# 將org3.json添加到config.json
# 利用jq指令將org3.json添加到config.json，並產出modified_config.json
# config.json包含Org1與Org2配置，modified_config.json包含Org1、Org2與Org3配置
# 後續將對config.json與modified_config.json重新轉換並計算出差異部份

$ jq -s '.[0] * {"channel_group":{"groups":{"Application":{"groups": {"Org3MSP":.[1]}}}}}' ./channel-artifacts/config.json ./organizations/peerOrganizations/org3.example.com/org3.json > ./channel-artifacts/modified_config.json

# 轉換2個json檔案並計算出差異部份
# 轉換config.json為protobuf檔案config.pb
$ configtxlator proto_encode --input ./channel-artifacts/config.json --type common.Config --output ./channel-artifacts/config.pb

# 轉換modified_config.json為protobuf檔案modified_config.pb
$ configtxlator proto_encode --input ./channel-artifacts/modified_config.json --type common.Config --output ./channel-artifacts/modified_config.pb

# 計算出2個protobuf檔案的差異
# 計算後產出protobuf檔案org3_update.pb
# org3_update.pb包含Org3定義與Org1、Org2的高階指向
$ configtxlator compute_update --channel_id dev-channel --original ./channel-artifacts/config.pb --updated ./channel-artifacts/modified_config.pb --output ./channel-artifacts/org3_update.pb

# 轉換org3_update.pb為JSON格式
$ configtxlator proto_decode --input ./channel-artifacts/org3_update.pb --type common.ConfigUpdate --output ./channel-artifacts/org3_update.json

# 封裝org3_update.json
# 封裝後產出org3_update_in_envelope.json
$ echo '{"payload":{"header":{"channel_header":{"channel_id":"'dev-channel'", "type":2}},"data":{"config_update":'$(cat ./channel-artifacts/org3_update.json)'}}}' | jq . > ./channel-artifacts/org3_update_in_envelope.json

# 轉換封裝檔為protobuf檔案
# 轉換後產出org3_update_in_envelope.pb
$ configtxlator proto_encode --input ./channel-artifacts/org3_update_in_envelope.json --type common.Envelope --output ./channel-artifacts/org3_update_in_envelope.pb

9.執行Channel更新

# Org1下執行簽章配置交易檔案org3_update_in_envelope.pb
# 此一步驟是在Org1下執行
$ peer channel signconfigtx -f ./channel-artifacts/org3_update_in_envelope.pb

# 轉換到Org2下執行更新Channel
$ export CORE_PEER_TLS_ENABLED=true
$ export CORE_PEER_LOCALMSPID="Org2MSP"
$ export CORE_PEER_TLS_ROOTCERT_FILE=${PWD}/organizations/peerOrganizations/org2.example.com/peers/peer0.org2.example.com/tls/ca.crt
$ export CORE_PEER_MSPCONFIGPATH=${PWD}/organizations/peerOrganizations/org2.example.com/users/Admin@org2.example.com/msp
$ export CORE_PEER_ADDRESS=localhost:9051

# 更新Channel
$ peer channel update -f ./channel-artifacts/org3_update_in_envelope.pb -c dev-channel -o localhost:7050 --ordererTLSHostnameOverride orderer.example.com --tls --cafile ${PWD}/organizations/ordererOrganizations/example.com/orderers/orderer.example.com/msp/tlscacerts/tlsca.example.com-cert.pem

10.Org3加入Channel

# Org3節點加入Channel
# 執行peer channel fetch 0取得創世區塊
# 執行peer channel fetch config取得更新後的配置區塊

# 轉換到Org3
$ export CORE_PEER_TLS_ENABLED=true
$ export CORE_PEER_LOCALMSPID="Org3MSP"
$ export CORE_PEER_TLS_ROOTCERT_FILE=${PWD}/organizations/peerOrganizations/org3.example.com/peers/peer0.org3.example.com/tls/ca.crt
$ export CORE_PEER_MSPCONFIGPATH=${PWD}/organizations/peerOrganizations/org3.example.com/users/Admin@org3.example.com/msp
$ export CORE_PEER_ADDRESS=localhost:11051

# 獲取創世區塊
# 若./channel-artifacts/dev-channel.block已存在，此步驟可略過。
$ peer channel fetch 0 ./channel-artifacts/dev-channel.block -o localhost:7050 --ordererTLSHostnameOverride orderer.example.com -c dev-channel --tls --cafile ${PWD}/organizations/ordererOrganizations/example.com/orderers/orderer.example.com/msp/tlscacerts/tlsca.example.com-cert.pem

# peer0.org3.example.com加入Channel
$ peer channel join -b ./channel-artifacts/dev-channel.block

11.設定Org3的錨節點

# peer0.org3.example.com設為Org3的錨節點
# 獲取dev-channel最新配置區塊並產生二進位的protobuf檔案
$ peer channel fetch config ./channel-artifacts/config_block.pb -o localhost:7050 --ordererTLSHostnameOverride orderer.example.com -c dev-channel --tls --cafile ${PWD}/organizations/ordererOrganizations/example.com/orderers/orderer.example.com/msp/tlscacerts/tlsca.example.com-cert.pem

# 轉換配置區塊為JSON格式與擷取配置內容
$ configtxlator proto_decode --input ./channel-artifacts/config_block.pb --type common.Block --output ./channel-artifacts/config_block.json
$ jq .data.data[0].payload.data.config ./channel-artifacts/config_block.json > ./channel-artifacts/config.json

# 利用jq指令將錨節點(anchor_peers)設定為peer0.org3.example.com並加入至Org3
# 執行後產出modified_anchor_config.json
$ jq '.channel_group.groups.Application.groups.Org3MSP.values += {"AnchorPeers":{"mod_policy": "Admins","value":{"anchor_peers": [{"host": "peer0.org3.example.com","port": 11051}]},"version": "0"}}' ./channel-artifacts/config.json > ./channel-artifacts/modified_anchor_config.json

# 轉換2個json檔案並計算出差異部份
# 轉換config.json為protobuf檔案config.pb
$ configtxlator proto_encode --input ./channel-artifacts/config.json --type common.Config --output ./channel-artifacts/config.pb

# 轉換modified_anchor_config.json為protobuf檔案modified_anchor_config.pb
$ configtxlator proto_encode --input ./channel-artifacts/modified_anchor_config.json --type common.Config --output ./channel-artifacts/modified_anchor_config.pb

# 計算出2個protobuf檔案的差異
# 計算後產出protobuf檔案anchor_update.pb
$ configtxlator compute_update --channel_id dev-channel --original ./channel-artifacts/config.pb --updated ./channel-artifacts/modified_anchor_config.pb --output ./channel-artifacts/anchor_update.pb

# 轉換anchor_update.pb為JSON格式
$ configtxlator proto_decode --input ./channel-artifacts/anchor_update.pb --type common.ConfigUpdate --output ./channel-artifacts/anchor_update.json

# 封裝anchor_update.json
$ echo '{"payload":{"header":{"channel_header":{"channel_id":"dev-channel", "type":2}},"data":{"config_update":'$(cat ./channel-artifacts/anchor_update.json)'}}}' | jq . > ./channel-artifacts/anchor_update_in_envelope.json

# 轉換封裝檔為protobuf檔案
$ configtxlator proto_encode --input ./channel-artifacts/anchor_update_in_envelope.json --type common.Envelope --output ./channel-artifacts/anchor_update_in_envelope.pb

# 更新Channel，設定peer0.org3.example.com為org3的錨節點
$ peer channel update -f ./channel-artifacts/anchor_update_in_envelope.pb -c dev-channel -o localhost:7050 --ordererTLSHostnameOverride orderer.example.com --tls --cafile ${PWD}/organizations/ordererOrganizations/example.com/orderers/orderer.example.com/msp/tlscacerts/tlsca.example.com-cert.pem
